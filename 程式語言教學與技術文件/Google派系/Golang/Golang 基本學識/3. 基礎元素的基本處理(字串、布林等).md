# 基礎元素的基本處理(字串、布林等)

## 字串

```go
func StringCompute() {
	fmt.Println("1" + "1")
	fmt.Println(len("Hello World")) //顯示字串長度
	fmt.Println("Hello World"[1])   //顯示"e" 的byte
	fmt.Println("Hello" + "World")

	a := "Hello World"
	fmt.Printf("%c", a[1]) //顯示e
	fmt.Print(a[1])
}
```

## 布林

```go
func BooleanCompute() {
	var a bool
	a = true
	fmt.Println("a = ", a)

	b := false
	fmt.Println("b =", b)

	fmt.Println(true && true)
	fmt.Println(true && false)
	fmt.Println(true || true)
	fmt.Println(true || false)
	fmt.Println(!true)
}
```
